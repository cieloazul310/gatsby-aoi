export const query = graphql`
  query {
    allMdxPost(sort: {fields: date, order: DESC}, limit: 4) {
      posts: edges {
        node {
          id
          title
          date(formatString: "YYYY-MM-DD")
          slug
          author {
            name
          }
        }
      }
    }
  }
`;
